Install

  • Windows

                
    New-Item -Path "HKLM:\SOFTWARE\Tailscale IPN"
    New-ItemProperty -Path 'HKLM:\Software\Tailscale IPN' -Name UnattendedMode -PropertyType String -Value always
    New-ItemProperty -Path 'HKLM:\Software\Tailscale IPN' -Name LoginURL -PropertyType String -Value https://YOUR-HEADSCALE-URL

  • Linux

    1. Register a machine (normal login)

                  
    headscale users create myfirstuser
    tailscale up --login-server http://10.106.107.31:8080
    headscale --user myfirstuser nodes register --key YOUR_MACHINE_KEY
    docker exec headscale headscale --user myfirstuser nodes register --key YOUR_MACHINE_KEY

    2. Register machine using a pre authenticated key

                          
    headscale --user myfirstuser preauthkeys create --reusable --expiration 24h
    tailscale up --login-server http://10.106.107.31:8080 --authkey 495bf29a4d352545458d96cdf2ac73cde2403cf5e102e7e1
